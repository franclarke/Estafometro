<task type="auto">
  <name>Task 1: Create initial schema migration</name>
  <files>supabase/migrations/001_initial_schema.sql</files>
  <action>
    Create supabase/migrations/ directory and the initial migration file with ALL tables defined in especificacion_tecnica.md section 4.

    **cases table:**
    ```sql
    CREATE TABLE cases (
      id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      public_id TEXT UNIQUE NOT NULL,
      status TEXT NOT NULL DEFAULT 'received' CHECK (status IN ('received', 'processing', 'requires_context', 'analyzed', 'partial', 'error', 'expired')),
      narrative_text TEXT,
      risk_level TEXT CHECK (risk_level IN ('low', 'medium', 'high', 'very_high')),
      risk_score INTEGER,
      final_summary TEXT,
      recommendations JSONB,
      created_at TIMESTAMPTZ DEFAULT now(),
      updated_at TIMESTAMPTZ DEFAULT now()
    );
    CREATE INDEX idx_cases_public_id ON cases(public_id);
    CREATE INDEX idx_cases_status ON cases(status);
    CREATE INDEX idx_cases_created_at ON cases(created_at);
    ```

    **case_evidence table:**
    ```sql
    CREATE TABLE case_evidence (
      id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      case_id UUID NOT NULL REFERENCES cases(id) ON DELETE CASCADE,
      evidence_type TEXT NOT NULL CHECK (evidence_type IN ('narrative', 'pasted_chat', 'screenshot', 'url', 'username', 'alias', 'phone', 'note')),
      content TEXT,
      storage_path TEXT,
      created_at TIMESTAMPTZ DEFAULT now()
    );
    CREATE INDEX idx_case_evidence_case_id ON case_evidence(case_id);
    ```

    **case_entities table:**
    ```sql
    CREATE TABLE case_entities (
      id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      case_id UUID NOT NULL REFERENCES cases(id) ON DELETE CASCADE,
      entity_type TEXT NOT NULL CHECK (entity_type IN ('platform', 'business_name', 'instagram_handle', 'url', 'domain', 'alias', 'cbu', 'phone', 'authority', 'bank', 'product', 'payment_method', 'marketplace')),
      value TEXT NOT NULL,
      normalized_value TEXT,
      created_at TIMESTAMPTZ DEFAULT now()
    );
    CREATE INDEX idx_case_entities_case_id ON case_entities(case_id);
    CREATE INDEX idx_case_entities_type ON case_entities(entity_type);
    ```

    **case_signals table:**
    ```sql
    CREATE TABLE case_signals (
      id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      case_id UUID NOT NULL REFERENCES cases(id) ON DELETE CASCADE,
      signal_code TEXT NOT NULL,
      signal_group TEXT NOT NULL,
      weight INTEGER NOT NULL,
      severity TEXT NOT NULL CHECK (severity IN ('low', 'medium', 'high', 'critical')),
      description TEXT,
      detected_at TIMESTAMPTZ DEFAULT now()
    );
    CREATE INDEX idx_case_signals_case_id ON case_signals(case_id);
    CREATE INDEX idx_case_signals_code ON case_signals(signal_code);
    ```

    **analysis_runs table:**
    ```sql
    CREATE TABLE analysis_runs (
      id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      case_id UUID NOT NULL REFERENCES cases(id) ON DELETE CASCADE,
      status TEXT NOT NULL DEFAULT 'pending' CHECK (status IN ('pending', 'running', 'completed', 'failed')),
      input_summary JSONB,
      extracted_data JSONB,
      signals_detected JSONB,
      subscores JSONB,
      final_score INTEGER,
      risk_level TEXT,
      confidence INTEGER,
      error_message TEXT,
      started_at TIMESTAMPTZ DEFAULT now(),
      completed_at TIMESTAMPTZ
    );
    CREATE INDEX idx_analysis_runs_case_id ON analysis_runs(case_id);
    ```

    **external_checks table:**
    ```sql
    CREATE TABLE external_checks (
      id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      case_id UUID NOT NULL REFERENCES cases(id) ON DELETE CASCADE,
      check_type TEXT NOT NULL,
      status TEXT NOT NULL CHECK (status IN ('pending', 'running', 'completed', 'failed')),
      result_summary TEXT,
      result_json JSONB,
      signal_impact JSONB,
      started_at TIMESTAMPTZ DEFAULT now(),
      completed_at TIMESTAMPTZ
    );
    CREATE INDEX idx_external_checks_case_id ON external_checks(case_id);
    ```

    **patterns table:**
    ```sql
    CREATE TABLE patterns (
      id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      code TEXT UNIQUE NOT NULL,
      name TEXT NOT NULL,
      category TEXT NOT NULL,
      summary TEXT NOT NULL,
      core_signals JSONB NOT NULL,
      high_weight_signals JSONB,
      hard_rules JSONB,
      counter_signals JSONB,
      variant_examples JSONB,
      recommended_actions JSONB NOT NULL,
      minimum_risk_level TEXT NOT NULL,
      version INTEGER NOT NULL DEFAULT 1,
      is_active BOOLEAN DEFAULT true,
      created_at TIMESTAMPTZ DEFAULT now(),
      updated_at TIMESTAMPTZ DEFAULT now()
    );
    CREATE INDEX idx_patterns_code ON patterns(code);
    CREATE INDEX idx_patterns_category ON patterns(category);
    ```

    **pattern_versions table:**
    ```sql
    CREATE TABLE pattern_versions (
      id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      pattern_id UUID NOT NULL REFERENCES patterns(id) ON DELETE CASCADE,
      version INTEGER NOT NULL,
      content JSONB NOT NULL,
      created_at TIMESTAMPTZ DEFAULT now(),
      UNIQUE(pattern_id, version)
    );
    ```

    **pattern_matches table:**
    ```sql
    CREATE TABLE pattern_matches (
      id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      case_id UUID NOT NULL REFERENCES cases(id) ON DELETE CASCADE,
      pattern_id UUID NOT NULL REFERENCES patterns(id) ON DELETE CASCADE,
      match_score INTEGER NOT NULL,
      matched_signals JSONB NOT NULL,
      created_at TIMESTAMPTZ DEFAULT now()
    );
    CREATE INDEX idx_pattern_matches_case_id ON pattern_matches(case_id);
    ```

    **candidate_patterns table:**
    ```sql
    CREATE TABLE candidate_patterns (
      id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      fingerprint TEXT UNIQUE NOT NULL,
      actor TEXT,
      threat TEXT,
      requested_action TEXT,
      payment_reason TEXT,
      theme TEXT,
      urgency TEXT,
      occurrence_count INTEGER DEFAULT 1,
      first_seen_at TIMESTAMPTZ DEFAULT now(),
      last_seen_at TIMESTAMPTZ DEFAULT now()
    );
    CREATE INDEX idx_candidate_patterns_fingerprint ON candidate_patterns(fingerprint);
    ```

    **case_candidate_pattern_links table:**
    ```sql
    CREATE TABLE case_candidate_pattern_links (
      id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      case_id UUID NOT NULL REFERENCES cases(id) ON DELETE CASCADE,
      candidate_pattern_id UUID NOT NULL REFERENCES candidate_patterns(id) ON DELETE CASCADE,
      created_at TIMESTAMPTZ DEFAULT now(),
      UNIQUE(case_id, candidate_pattern_id)
    );
    ```

    **feedback table:**
    ```sql
    CREATE TABLE feedback (
      id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      case_id UUID NOT NULL REFERENCES cases(id) ON DELETE CASCADE,
      helpful BOOLEAN NOT NULL,
      comment TEXT,
      created_at TIMESTAMPTZ DEFAULT now()
    );
    CREATE INDEX idx_feedback_case_id ON feedback(case_id);
    ```

    **analytics_events table:**
    ```sql
    CREATE TABLE analytics_events (
      id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      event_name TEXT NOT NULL,
      case_id UUID REFERENCES cases(id) ON DELETE SET NULL,
      payload JSONB,
      created_at TIMESTAMPTZ DEFAULT now()
    );
    CREATE INDEX idx_analytics_events_name ON analytics_events(event_name);
    CREATE INDEX idx_analytics_events_case_id ON analytics_events(case_id);
    ```

    **signal_catalog table:**
    ```sql
    CREATE TABLE signal_catalog (
      id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
      code TEXT UNIQUE NOT NULL,
      group_name TEXT NOT NULL,
      description TEXT NOT NULL,
      default_weight INTEGER NOT NULL,
      severity TEXT NOT NULL CHECK (severity IN ('low', 'medium', 'high', 'critical')),
      is_active BOOLEAN DEFAULT true,
      created_at TIMESTAMPTZ DEFAULT now()
    );
    CREATE INDEX idx_signal_catalog_code ON signal_catalog(code);
    CREATE INDEX idx_signal_catalog_group ON signal_catalog(group_name);
    ```

    **Enable RLS on all tables** and create restrictive policies (cases are only readable by owner via public_id for MVP, no auth needed — use case-by-case RLS skip for service role).

    **Add updated_at trigger** for all tables with updated_at column.
  </action>
