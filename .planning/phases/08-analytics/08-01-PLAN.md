  <name>Task 3: Create SQL views for metrics</name>
  <files>supabase/migrations/003_analytics_views.sql</files>
  <action>
    Create SQL migration with views for dashboard metrics:

    **cases_metrics view:**
    ```sql
    CREATE VIEW cases_metrics AS
    SELECT
      date_trunc('day', created_at) as day,
      status,
      risk_level,
      COUNT(*) as count
    FROM cases
    GROUP BY day, status, risk_level
    ORDER BY day DESC
    ```

    **signals_metrics view:**
    ```sql
    CREATE VIEW signals_metrics AS
    SELECT
      signal_code,
      signal_group,
      severity,
      COUNT(*) as count
    FROM case_signals
    GROUP BY signal_code, signal_group, severity
    ORDER BY count DESC
    ```

    **patterns_metrics view:**
    ```sql
    CREATE VIEW patterns_metrics AS
    SELECT
      p.code,
      p.name,
      COUNT(pm.id) as match_count,
      AVG(pm.match_score) as avg_match_score
    FROM patterns p
    LEFT JOIN pattern_matches pm ON p.id = pm.pattern_id
    GROUP BY p.code, p.name
    ORDER BY match_count DESC
    ```

    **feedback_metrics view:**
    ```sql
    CREATE VIEW feedback_metrics AS
    SELECT
      date_trunc('day', created_at) as day,
      helpful,
      COUNT(*) as count
    FROM feedback
    GROUP BY day, helpful
    ORDER BY day DESC
    ```

    **funnel_metrics (SQL function):**
    ```sql
    CREATE FUNCTION get_funnel_metrics(start_date DATE, end_date DATE)
    RETURNS TABLE(stage TEXT, count BIGINT) AS $$
    BEGIN
      RETURN QUERY
      SELECT 'started' as stage, COUNT(*) FROM cases WHERE created_at >= start_date AND created_at <= end_date
      UNION ALL
      SELECT 'analyzed', COUNT(*) FROM cases WHERE status = 'analyzed' AND created_at >= start_date AND created_at <= end_date
      UNION ALL
      SELECT 'feedback', COUNT(*) FROM feedback f JOIN cases c ON f.case_id = c.id WHERE f.created_at >= start_date AND f.created_at <= end_date;
    END
    $$ LANGUAGE plpgsql
    ```
