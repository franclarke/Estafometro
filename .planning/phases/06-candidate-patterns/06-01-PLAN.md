    interface StructuralSignature {
      actor: string       // 'authority' | 'stranger' | 'family' | 'business' | 'unknown'
      threat: string      // 'arrest' | 'financial_loss' | 'account_loss' | 'none' | 'unknown'
      requested_action: string  // 'transfer_money' | 'share_data' | 'call_back' | 'none' | 'unknown'
      payment_reason: string    // 'avoid_consequences' | 'purchase' | 'help' | 'none' | 'unknown'
      theme: string       // 'minor_case' | 'prize' | 'technical' | 'urgent_family' | 'marketplace' | 'unknown'
      urgency: string     // 'high' | 'medium' | 'low'
    }

    export function buildSignature(extraction: ExtractionResult, signals: CaseSignal[]): StructuralSignature {
      // Derive actor from entities and signals:
      //   - authority_impersonation signals → actor = 'authority'
      //   - family signals → actor = 'family'
      //   - business signals → actor = 'business'
      //   - otherwise → actor = 'stranger' or 'unknown'

      // Derive threat from signals:
      //   - threatens_arrest → threat = 'arrest'
      //   - asks_for_credentials/otp → threat = 'account_loss'
      //   - transfer_request → threat = 'financial_loss'

      // Derive requested_action from extraction.requested_action

      // Derive payment_reason from context:
      //   - bribery_request → payment_reason = 'avoid_consequences'
      //   - online_purchase case type → payment_reason = 'purchase'

      // Derive theme from case_type and narrative_theme

      // Derive urgency from signals:
      //   - urgent_transfer + emotional_pressure → urgency = 'high'
      //   - transfer_request alone → urgency = 'medium'
    }

    export function hashFingerprint(signature: StructuralSignature): string {
      // Serialize to JSON, normalize keys alphabetically
      // SHA-256 hash, return first 16 chars as hex
      return crypto.createHash('sha256').update(JSON.stringify(signature, Object.keys(signature).sort())).digest('hex').slice(0, 16)
    }
